def full_justify(words: list[str], max_width: int) -> list[str]:
    def justify_line(line, width):
        if len(line) == 1:
            return line[0] + ' ' * (max_width - len(line[0]))
        
        total_spaces = max_width - sum(len(word) for word in line)
        space_gaps = len(line) - 1
        spaces_per_gap, extra_spaces = divmod(total_spaces, space_gaps)
        
        justified_line = []
        for i, word in enumerate(line):
            if i == 0:
                justified_line.append(word)
            else:
                justified_line.append(' ' * (spaces_per_gap + (1 if extra_spaces > 0 else 0)))
                extra_spaces -= 1
                justified_line.append(word)
        
        return ''.join(justified_line)
