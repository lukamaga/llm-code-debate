        """Add an interval to the manager."""
        for i in range(len(self.intervals)):
            if start <= self.intervals[i][1] and end >= self.intervals[i][0]:
                start = min(start, self.intervals[i][0])
                end = max(end, self.intervals[i][1])

from typing import List, Tuple

class IntervalManager:
    def __init__(self):
        self.intervals: List[Tuple[int, int]] = []

    def add_interval(self, start: int, end: int) -> None:
        # Merge overlapping intervals
        new_intervals = []
        for interval in self.intervals:
            if interval[1] < start or interval[0] > end:
                new_intervals.append(interval)
            else:
                start = min(start, interval[0])
                end = max(end, interval[1])
        new_intervals.append((start, end))
        # Sort intervals by start time
        self.intervals = sorted(new_intervals, key=lambda x: x[0])

    def get_merged(self) -> List[Tuple[int, int]]:
        return self.intervals

    def is_covered(self, point: int) -> bool:
        for interval in self.intervals:
            if interval[0] <= point <= interval[1]:
                return True
        return False

    def min_hitting_set(self) -> List[int]:
        hitting_set = []
        for interval in self.intervals:
            # Find the midpoint of each interval and add to hitting set
            midpoint = (interval[0] + interval[1]) // 2
            if midpoint not in hitting_set:
                hitting_set.append(midpoint)
        return hitting_set
