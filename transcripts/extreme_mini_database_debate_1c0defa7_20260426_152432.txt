class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transaction_stack = []

    def get(self, key: str) -> str | None:
        if not self.transaction_stack:
            return self.storage.get(key)
        else:
            for transaction in reversed(self.transaction_stack):
                if key in transaction['changes']:
                    return transaction['changes'][key]
            return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self.transaction_stack:
            self.storage.set(key, value)
        else:
            top_transaction = self.transaction_stack[-1]
            top_transaction['changes'][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:
            return self.storage.delete(key)
        else:
            top_transaction = self.transaction_stack[-1]
            if key in top_transaction['changes']:
                del top_transaction['changes'][key]
                return True
            elif self.storage.has(key):
                top_transaction['changes'][key] = None
                return True

            else:
                self.storage.set(key, value)
