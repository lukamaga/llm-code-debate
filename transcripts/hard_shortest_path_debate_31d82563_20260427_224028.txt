        if u not in self.graph:
            self.graph[u] = {}

            self.graph[v] = {}
        self.graph[u][v] = weight

    def shortest_path(self, start: str, end: str) -> tuple[float, list[str]]:
        distances = {node: float('inf') for node in self.graph}
        distances[start] = 0
        previous = {node: None for node in self.graph}
        unvisited = set(self.graph)

        while unvisited:
            current = min(unvisited, key=lambda node: distances[node])
            unvisited.remove(current)

            for neighbor, weight in self.graph[current].items():

                    previous[neighbor] = current

        path = [end]

        return (distances[end], path)

from heapq import heappop, heappush
from typing import Dict, List, Tuple

class WeightedGraph:
    def __init__(self):
        self.graph: Dict[str, Dict[str, float]] = {}

    def add_edge(self, u: str, v: str, weight: float) -> None:
        if u not in self.graph:
            self.graph[u] = {}
        self.graph[u][v] = weight

    def shortest_path(self, start: str, end: str) -> Tuple[float, List[str]]:
        distances = {node: float('inf') for node in self.graph}
        previous_nodes = {node: None for node in self.graph}
        distances[start] = 0
        heap = [(0, start)]

        while heap:
            current_distance, current_node = heappop(heap)

            if current_distance > distances[current_node]:
                continue

            for neighbor, weight in self.graph.get(current_node, {}).items():
                distance = current_distance + weight

                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    previous_nodes[neighbor] = current_node
                    heappush(heap, (distance, neighbor))

        path, current_node = [], end
        while current_node is not None:
            path.insert(0, current_node)
            current_node = previous_nodes[current_node]

        if len(path) == 1 and start != end:
            return float('inf'), []

        return distances[end], path

    def all_shortest_from(self, start: str) -> Dict[str, float]:
        distances = {node: float('inf') for node in self.graph}
        distances[start] = 0
        heap = [(0, start)]

        while heap:
            current_distance, current_node = heappop(heap)

            if current_distance > distances[current_node]:
                continue

            for neighbor, weight in self.graph.get(current_node, {}).items():
                distance = current_distance + weight

                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    heappush(heap, (distance, neighbor))

        return {node: distances[node] for node in distances if distances[node] != float('inf')}

    def has_path(self, start: str, end: str) -> bool:
        _, path = self.shortest_path(start, end)
        return end in path

import heapq
from typing import List, Tuple, Dict, Union

class WeightedGraph:
    def __init__(self):
        # Graph is a dictionary where keys are nodes and values are lists of tuples (node, weight)
        self.graph = {}

    def add_edge(self, u: str, v: str, weight: float) -> None:
        if u not in self.graph:
            self.graph[u] = []
        self.graph[u].append((v, weight))

    def shortest_path(self, start: str, end: str) -> Tuple[Union[int, float], List[str]]:
        heap = [(0, start, [])]  # (distance, node, path)
        shortest_distances = {start: 0}  # keep track of shortest distances from start to all other nodes
        visited = set()

        while heap:
            (dist, current, path) = heapq.heappop(heap)
            if current in visited:
                continue
            visited.add(current)
            path = path + [current]

            # if we reached the end node, return distance and path
            if current == end:
                return dist, path

            for neighbor, weight in self.graph.get(current, []):
                if neighbor not in visited:
                    old_dist = shortest_distances.get(neighbor, None)
                    new_dist = dist + weight
                    if old_dist is None or new_dist < old_dist:
                        shortest_distances[neighbor] = new_dist
                        heapq.heappush(heap, (new_dist, neighbor, path))

        # if no path exists from start to end
        return float('inf'), []

    def all_shortest_from(self, start: str) -> Dict[str, float]:
        return self.shortest_path(start, None)[1]

    def has_path(self, start: str, end: str) -> bool:
        dist, path = self.shortest_path(start, end)
        return dist != float('inf')

import heapq
from typing import List, Tuple, Dict, Union

class WeightedGraph:
    def __init__(self):
        # Graph is a dictionary where keys are nodes and values are lists of tuples (node, weight)
        self.graph = {}

    def add_edge(self, u: str, v: str, weight: float) -> None:
        if u not in self.graph:
            self.graph[u] = []
        self.graph[u].append((v, weight))

    def shortest_path(self, start: str, end: str) -> Tuple[Union[int, float], List[str]]:
        heap = [(0, start, [])]  # (distance, node, path)
        shortest_distances = {start: 0}  # keep track of shortest distances from start to all other nodes
        visited = set()

        while heap:
            (dist, current, path) = heapq.heappop(heap)
            if current in visited:
                continue
            visited.add(current)
            path = path + [current]

            # if we reached the end node, return distance and path
            if current == end:
                return dist, path

            for neighbor, weight in self.graph.get(current, []):
                if neighbor not in visited:
                    old_dist = shortest_distances.get(neighbor, None)
                    new_dist = dist + weight
                    if old_dist is None or new_dist < old_dist:
                        shortest_distances[neighbor] = new_dist
                        heapq.heappush(heap, (new_dist, neighbor, path))

        # if no path exists from start to end
        return float('inf'), []

    def all_shortest_from(self, start: str) -> Dict[str, float]:
        return self.shortest_path(start, None)[1]

    def has_path(self, start: str, end: str) -> bool:
        dist, path = self.shortest_path(start, end)
        return dist != float('inf')
