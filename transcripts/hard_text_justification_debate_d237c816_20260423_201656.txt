from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start].ljust(max_width)
        
        num_words = end - start + 1
        spaces_needed = max_width - total_length
        spaces_between_words = [0] * (num_words - 1)
        extra_spaces = spaces_needed
        
        for i in range(num_words - 2, -1, -1):
            spaces_between_words[i] = spaces_needed // (num_words - 1) + (extra_spaces > 0)
            extra_spaces -= spaces_between_words[i]
        
        line = words[start]
        for i in range(1, num_words):
            line += ' ' * spaces_between_words[i-1] + words[start + i]
        
        return line.ljust(max_width)

    result = []
    start = 0
    total_length = 0
    
    for end, word in enumerate(words):
        if total_length + len(word) + (end - start) > max_width:
            result.append(justify_line(words, start, end-1, total_length))
            start = end
            total_length = len(word)
        else:
            total_length += len(word)
    
    # Handle the last line which should be left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line.ljust(max_width))
    
    return result

def full_justify(words, max_width):
    result = []
    i = 0
    while i < len(words):
        total_length = len(words[i])
        j = i + 1
        while j < len(words) and (total_length + len(words[j]) + j - i - 1 < max_width):
            total_length += len(words[j])
            j += 1
            
        num_spaces = max_width - total_length
        if j == len(words) or j - i == 1: # left justify the last line or one word per line
            line = ' '.join(words[i:j]).ljust(max_width)
            result.append(line[:max_width])
        else: 
            num_spaces_between = num_spaces // (j - i - 1)
            num_extra_spaces = num_spaces % (j - i - 1)
            line = ''
            for k in range(i, j):
                if k != i: # add extra spaces between words
                    line += ' ' * num_spaces_between

                        line += ' '
                        num_extra_spaces -= 1
                line += words[k]
            result.append(line)
        i = j
    return result
