from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start].ljust(max_width)
        
        num_words = end - start + 1
        spaces_needed = max_width - total_length
        spaces_between = [0] * (num_words - 1)
        extra_spaces = spaces_needed % (num_words - 1)
        
        for i in range(num_words - 1):
            spaces_between[i] += (spaces_needed // (num_words - 1)) + (1 if i < extra_spaces else 0)
        
        return (' ' * spaces_between[0]).join(words[start]) + ''.join((' ' * s).join(w) for w, s in zip(words[start+1:end], spaces_between[1:]))

    result = []
    start = 0
    total_length = 0
    
    for i, word in enumerate(words):
        if total_length + len(word) + (i - start) > max_width:
            result.append(justify_line(words, start, i-1, total_length))
            start = i
            total_length = len(word)
        else:
            total_length += len(word)
    
    # Last line left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result

# Example usage:

def full_justify(words, max_width):
    result = []
    i = 0
    
    while i < len(words):
        total_length = len(words[i])
        next_line_length = -1  # account for the space between words
        num_of_words = 1  
        
        while (i + 1 < len(words)) and ((total_length + len(words[i+1]) + num_of_words) <= max_width):
            i += 1
            total_length += len(words[i])
            next_line_length = total_length  # for last line, we don't need to add extra spaces
            num_of_words += 1  
            
        if num_of_words == 1:  # only one word in the current line or a single word fits into the maxWidth
            result.append(words[i].ljust(max_width))
        elif i == len(words) - 1:  # last line, left-justified
            result.append(' '.join(words[num_of_words-1:]).ljust(max_width))
        else:  
            each_extra_space = (max_width - total_length) // (num_of_words - 1)
            extra_spaces = (max_width - total_length) % (num_of_words - 1)
            
            for j in range(each_extra_space):
                words[j] += ' '
                
            for k in range(extra_spaces):
                words[k] += ' '
        
            result.append(''.join(words))
        i += 1
            
    return result

def full_justify(words: list[str], max_width: int) -> list[str]:
    """
    Given an array of strings words and a width maxWidth, format the text such that each line has exactly maxWidth characters and is fully (left and right) justified. Pack as many words as you can in each line. Pad extra spaces when necessary. Extra spaces between words should be distributed as evenly as possible. The last line should be left-justified.
    """
    lines = []
    current_line = ""
    for word in words:
        if len(current_line) + len(word) > max_width:
            lines.append(current_line)
            current_line = word
        else:
            current_line += " " + word
    lines.append(current_line)
    return lines

from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start].ljust(max_width)
        
        num_words = end - start + 1
        spaces_needed = max_width - total_length
        spaces_between = [0] * (num_words - 1)
        extra_spaces = spaces_needed % (num_words - 1)
        
        for i in range(num_words - 1):
            spaces_between[i] += (spaces_needed // (num_words - 1)) + (1 if i < extra_spaces else 0)
        
        return (' ' * spaces_between[0]).join(words[start]) + ''.join((' ' * s).join(w) for w, s in zip(words[start+1:end], spaces_between[1:]))

    result = []
    start = 0
    total_length = 0
    
    for i, word in enumerate(words):
        if total_length + len(word) + (i - start) > max_width:
            result.append(justify_line(words, start, i-1, total_length))
            start = i
            total_length = len(word)
        else:
            total_length += len(word)
    
    # Last line left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result

from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start].ljust(max_width)
        
        num_words = end - start + 1
        spaces_needed = max_width - total_length
        spaces_between = [0] * (num_words - 1)
        extra_spaces = spaces_needed % (num_words - 1)
        
        for i in range(num_words - 1):
            spaces_between[i] += (spaces_needed // (num_words - 1)) + (1 if i < extra_spaces else 0)
        
        return (' ' * spaces_between[0]).join(words[start]) + ''.join((' ' * s).join(w) for w, s in zip(words[start+1:end], spaces_between[1:]))

    result = []
    start = 0
    total_length = 0
    
    for i, word in enumerate(words):
        if total_length + len(word) + (i - start) > max_width:
            result.append(justify_line(words, start, i-1, total_length))
            start = i
            total_length = len(word)
        else:
            total_length += len(word)
    
    # Last line left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result
