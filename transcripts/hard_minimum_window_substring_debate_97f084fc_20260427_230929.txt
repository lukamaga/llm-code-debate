def min_window(s, t):
    if not s or not t:
        return ""

    need = {char: count for char, count in zip(t, [1] * len(t))}
    window = {}
    left, right = 0, 0
    valid = 0
    start, end = float('inf'), float('inf')
    while right < len(s):
        char = s[right]
        right += 1

        if char in need:
            window[char] = window.get(char, 0) + 1
            if window[char] <= need[char]:
                valid += 1

        while valid == len(t):
            if right - left < end - start:
                start, end = left, right
            char = s[left]
            left += 1

            if char in need:
                if window[char] <= need[char]:
                    valid -= 1
                window[char] -= 1

    return "" if start == float('inf') else s[start:end]

def min_window(s: str, t: str) -> str:
    from collections import Counter

    if not t or len(t) > len(s):  # if t is empty or longer than s
        return ""

    freq_map = Counter(t)  # frequency map of characters in t
    required = len(freq_map)  # number of unique characters required

    l, r = 0, 0  # pointers for the sliding window
    formed = 0  # count of unique characters formed in current window
    window_counts = {}  # frequency map of characters in current window

    ans = float('inf'), None, None  # initialize answer as infinity and two None values

    while r < len(s):  # while the right pointer is within bounds
        character = s[r]  # get character at right pointer
        window_counts[character] = window_counts.get(character, 0) + 1  # increment count in map

        if character in freq_map and window_counts[character] == freq_map[character]:  # if character is required and counts match
            formed += 1  # increment formed count

        while l <= r and formed == required:  # while left pointer is valid and all characters are formed
            character = s[l]  # get character at left pointer

            if r - l + 1 < ans[0]:  # if current window length is less than recorded minimum
                ans = (r - l + 1, l, r)  # update answer to new minimum and save pointers

            window_counts[character] -= 1  # decrement count in map as we slide the window
            if character in freq_map and window_counts[character] < freq_map[character]:  # if character is required and counts don't match
                formed -= 1  # decrement formed count as current window no longer has all characters formed

            l += 1  # move the left pointer to right by one

        r += 1  # move the right pointer to right by one
    return "" if ans[0] == float('inf') else s[ans[1]:ans[2]+1]  # return empty string or minimum window substring
