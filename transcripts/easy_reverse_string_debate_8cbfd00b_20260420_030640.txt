        s (list[str]): The input list of characters to be reversed.

    Returns:
        None: Modifies the input list in-place.

    Raises:
        ValueError: If the input list is empty or contains non-printable ASCII characters.

        raise ValueError("Input list cannot be empty")

        s[i], s[-1 - i] = s[-1 - i], s[i]

def reverse_string(s: list[str]) -> None:
    start, end = 0, len(s) - 1
    while start < end:
        s[start], s[end] = s[end], s[start]
        start += 1
        end -= 1
