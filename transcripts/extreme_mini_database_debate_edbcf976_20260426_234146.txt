from typing import List, Dict, Tuple
from storage import StorageBackend

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transaction_stack = []

    def get(self, key: str) -> str | None:
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self.transaction_stack:
            self.storage.set(key, value)
        else:
            self.transaction_stack[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:
            return self.storage.delete(key)
        else:
            self.transaction_stack[-1][key] = None
            return True

    def begin(self) -> None:
        self.transaction_stack.append({})

    def commit(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        for key, value in self.transaction_stack[-1].items():
            if value is None:
                self.storage.delete(key)
            else:
                self.storage.set(key, value)

        self.transaction_stack.pop()

    def rollback(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        snapshot = self.storage.snapshot()
        for key, _ in self.transaction_stack[-1].items():
            self.storage.delete(key)

        self.transaction_stack.pop()
        self.storage.restore(snapshot)

from storage import StorageBackend

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transactions = []

    def get(self, key: str) -> str | None:
        return self._get_current_transaction_value(key)

    def set(self, key: str, value: str) -> None:
        self._set_current_transaction_value(key, value)

    def delete(self, key: str) -> bool:
        if self.storage.has(key):
            self._delete_current_transaction_value(key)
            return True
        return False

    def begin(self) -> None:
        self.transactions.append({})

    def commit(self) -> None:
        if not self.transactions:
            raise Exception("No active transaction")
        current_transaction = self.transactions.pop()
        for key, value in current_transaction.items():
            if value is None:
                self.storage.delete(key)
            else:
                self.storage.set(key, value)

    def rollback(self) -> None:
        if not self.transactions:
            raise Exception("No active transaction")
        self.transactions.pop()

    def _get_current_transaction_value(self, key: str) -> str | None:
        for transaction in reversed(self.transactions):
            if key in transaction:
                return transaction[key]
        return self.storage.get(key)

    def _set_current_transaction_value(self, key: str, value: str) -> None:
        current_transaction = self.transactions[-1] if self.transactions else {}
        current_transaction[key] = value

    def _delete_current_transaction_value(self, key: str) -> None:
        current_transaction = self.transactions[-1] if self.transactions else {}
        current_transaction[key] = None

from storage import StorageBackend

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transaction_stack = []

    def get(self, key: str) -> str | None:
        if self.transaction_stack:
            for stack in reversed(self.transaction_stack):
                if key in stack:
                    return stack[key] if stack[key] is not None else None
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self.transaction_stack:
            self.storage.set(key, value)
        else:
            self.transaction_stack[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:
            return self.storage.delete(key)
        else:
            self.transaction_stack[-1][key] = None
            return True

    def begin(self) -> None:
        self.transaction_stack.append({})

    def commit(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        for key, value in self.transaction_stack[-1].items():
            if value is None:
                self.storage.delete(key)
            else:
                self.storage.set(key, value)

        self.transaction_stack.pop()

    def rollback(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        snapshot = self.storage.snapshot()
        for key, _ in self.transaction_stack[-1].items():
            self.storage.delete(key)

        self.transaction_stack.pop()
        self.storage.restore(snapshot)

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transaction_stack = []

    def get(self, key: str) -> str | None:
        if self.transaction_stack:
            for stack in reversed(self.transaction_stack):
                if key in stack:
                    return stack[key] if stack[key] is not None else None
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self.transaction_stack:
            self.storage.set(key, value)
        else:
            self.transaction_stack[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:
            return self.storage.delete(key)
        else:
            self.transaction_stack[-1][key] = None
            return True

    def begin(self) -> None:
        # Create a deep copy of the current transaction state
        self.transaction_stack.append(copy.deepcopy(self.storage.data))

    def commit(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        for key, value in self.transaction_stack[-1].items():
            # Check if the key exists before committing it
            if value is None and self.storage.has(key):
                self.storage.delete(key)
            elif not self.storage.has(key) or (value is not None and self.storage.get(key) != value):
                self.storage.set(key, value)

        self.transaction_stack.pop()

    def rollback(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        snapshot = self.storage.snapshot()
        for key in self.transaction_stack[-1].keys():
            # Restore the original state of each key from the snapshot, if it exists
            if key in snapshot:
                self.storage.set(key, snapshot[key])
            else:

    def get(self, key: str) -> str | None:
        if self.transaction_stack:
            for stack in reversed(self.transaction_stack):
                if key in stack:
                    return stack[key] if stack[key] is not None else None

        if not self.transaction_stack:
            self.storage.set(key, value)

            self.transaction_stack[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:

        else:
            self.transaction_stack[-1][key] = None
            return True

        self.transaction_stack.append({})

from storage import StorageBackend

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transaction_stack = []

    def get(self, key: str) -> str | None:
        if self.transaction_stack:
            for stack in reversed(self.transaction_stack):
                if key in stack:
                    return stack[key] if stack[key] is not None else None
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self.transaction_stack:
            self.storage.set(key, value)
        else:
            self.transaction_stack[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self.transaction_stack:
            return self.storage.delete(key)
        else:
            self.transaction_stack[-1][key] = None
            return True

    def begin(self) -> None:
        self.transaction_stack.append({})

    def commit(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        for key, value in self.transaction_stack[-1].items():
            if value is None:
                self.storage.delete(key)
            else:
                self.storage.set(key, value)

        self.transaction_stack.pop()

    def rollback(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')

        snapshot = self.storage.snapshot()
        for key, _ in self.transaction_stack[-1].items():
            self.storage.delete(key)

        self.transaction_stack.pop()
        self.storage.restore(snapshot)
