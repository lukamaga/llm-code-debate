class IntervalManager:
    def __init__(self):
        self.intervals = []  # sorted list of disjoint intervals

    def add_interval(self, start: int, end: int) -> None:
        if not self.intervals or self.intervals[-1][1] < start:
            self.intervals.append((start, end))
        elif self.intervals[-1][0] <= start <= self.intervals[-1][1]:
            self.intervals[-1] = (self.intervals[-1][0], max(end, self.intervals[-1][1]))
        else:  # binary search for correct position to insert new interval
            lo, hi = 0, len(self.intervals) - 1
            while lo <= hi:
                mid = (lo + hi) // 2
                if self.intervals[mid][1] < start:
                    lo = mid + 1
                elif self.intervals[mid][0] > end:
                    hi = mid - 1
                else:  # overlap with existing interval found, merge and sort again
                    start = min(self.intervals[mid][0], start)
                    end = max(self.intervals[mid][1], end)
                    del self.intervals[mid]
                    break
            self.intervals.insert(lo, (start, end))  # insert new merged interval at correct position
        # merge intervals again in case of overlap with inserted interval
        i = 0
        while i < len(self.intervals) - 1:
            if self.intervals[i][1] >= self.intervals[i + 1][0]:
                self.intervals[i] = (self.intervals[i][0], max(self.intervals[i][1], self.intervals[i + 1][1]))
                del self.intervals[i + 1]
            else:
                i += 1

    def get_merged(self) -> list[tuple[int, int]]:
        return self.intervals

    def is_covered(self, point: int) -> bool:
        lo, hi = 0, len(self.intervals) - 1
        while lo <= hi:
            mid = (lo + hi) // 2
            if self.intervals[mid][1] < point:
                lo = mid + 1
            elif self.intervals[mid][0] > point:
                hi = mid - 1
            else:  # interval found that covers the point
                return True
        return False

    def min_hitting_set(self) -> list[int]:
        points = []
        for start, end in self.intervals:
            if not points or points[-1] < start:
                points.append(start)
            elif points[-1] <= end:
                points[-1] = max(points[-1], start)
        return points

class IntervalManager:
    def __init__(self):
        self.intervals = []

    def add_interval(self, start: int, end: int) -> None:
        if start > end:
            raise ValueError("Invalid interval")
        new_intervals = []
        for interval in self.intervals:
            if interval[1] < start or interval[0] > end:
                new_intervals.append(interval)
            else:
                start = min(start, interval[0])
                end = max(end, interval[1])
        new_intervals.append((start, end))
        self.intervals = sorted(new_intervals, key=lambda x: x[0])

    def get_merged(self) -> list[tuple[int, int]]:
        return sorted(self.intervals, key=lambda x: x[0])

    def is_covered(self, point: int) -> bool:
        for interval in self.intervals:
            if interval[0] <= point <= interval[1]:
                return True
        return False

    def min_hitting_set(self) -> list[int]:
        covered_points = set()
        for interval in self.intervals:
            for point in range(interval[0], interval[1]+1):
                covered_points.add(point)
        return sorted(list(covered_points))
