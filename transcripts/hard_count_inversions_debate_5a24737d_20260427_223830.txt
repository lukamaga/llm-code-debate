def count_inversions(arr):
    def merge_and_count(left, right):
        inv_count = 0
        i, j = 0, 0

        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                arr[i + j] = left[i]
                i += 1
            else:
                arr[i + j] = right[j]
                j += 1
                inv_count += len(left) - i

        arr[i + j:] = left[i:]
        arr[i + j:] = right[j:]

        return inv_count

    def merge_sort(arr):
        if len(arr) <= 1:
            return arr, 0

        mid = len(arr) // 2
        left, left_inv_count = merge_sort(arr[:mid])
        right, right_inv_count = merge_sort(arr[mid:])
        merged, merged_inv_count = merge_and_count(left, right)
        return merged, left_inv_count + right_inv_count + merged_inv_count
