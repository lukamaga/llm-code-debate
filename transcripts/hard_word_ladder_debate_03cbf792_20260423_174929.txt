from collections import deque

def ladder_length(begin_word, end_word, word_list):
    wordList = set(word_list) # Convert list to a set for O(1) lookups
    queue = deque([[begin_word]]) 
    
    while queue:
        path = queue.popleft()
        node = path[-1]
        
        if node == end_word:
            return len(path) # Return the length of the path
            
        for i in range(len(node)): 
            for c in 'abcdefghijklmnopqrstuvwxyz': 
                next_word = node[:i]+c+node[i+1:]
                
                if next_word in wordList and next_word not in path: # Check if the new word is in our list of words and has not been visited before
                    queue.append(list(path) + [next_word]) 
                    
    return 0

def ladder_length(begin_word: str, end_word: str, word_list: List[str]) -> int:

    while queue:
        current_word, length = queue.popleft()

                    queue.append((next_word, length + 1))

from collections import deque
from typing import List

def ladder_length(begin_word: str, end_word: str, word_list: List[str]) -> int:
    if end_word not in word_list:
        return 0
    
    word_set = set(word_list)
    queue = deque([(begin_word, 1)])
    visited = {begin_word}
    
    while queue:
        current_word, length = queue.popleft()
        
        for i in range(len(current_word)):
            for char in 'abcdefghijklmnopqrstuvwxyz':
                next_word = current_word[:i] + char + current_word[i+1:]
                
                if next_word == end_word:
                    return length + 1
                
                if next_word in word_set and next_word not in visited:
                    queue.append((next_word, length + 1))
                    visited.add(next_word)
    
    return 0
