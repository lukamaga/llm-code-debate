from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start] + ' ' * (max_width - total_length)
        
        num_spaces = end - start
        spaces_between_words = (max_width - total_length) // num_spaces
        extra_spaces = (max_width - total_length) % num_spaces
        
        line = words[start]
        for i in range(1, num_spaces + 1):
            line += ' ' * (spaces_between_words + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    current_length = len(words[0])
    
    for i in range(1, len(words)):
        word_length = len(words[i])
        if current_length + 1 + word_length > max_width:
            result.append(justify_line(words, start, i - 1, current_length))
            start = i
            current_length = word_length
        else:
            current_length += 1 + word_length
    
    # Handle the last line which should be left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result

# Example usage:
words = ["This", "is", "an", "example", "of", "text", "justification."]

def full_justify(words, max_width):
    res = []
    i = 0
    while i < len(words):
        total_length = len(words[i])
        j = i + 1
        while j < len(words) and (total_length + len(words[j]) + j - i - 1 < max_width):
            total_length += len(words[j])
            j += 1
            
        word_count = j - i
        space_count = max_width - total_length
        
        if j == len(words) or word_count == 1:  # last line or one word per line, left justified
            line = ' '.join(words[i:j]) + ' ' * (max_width - len(line))
        else:  # normal lines with evenly distributed spaces
            space_div, space_mod = divmod(space_count, word_count - 1)
            line = ''.join([words[k] + ' ' * (space_div + (1 if k < i + space_mod else 0)) for k in range(i, j)]).rstrip()
            
        res.append(line)
        i = j
    return res

def full_justify(words, max_width):
    res = []
    i = 0
    while i < len(words):
        total_length = len(words[i])
        j = i + 1
        # Handle case where a line has only one word by left-justifying it
        if j == len(words) or total_length + len(' '.join(words[i:j])) > max_width:  
            res.append(words[i] + ' ' * (max_width - total_length))
            i = j
        else:  # normal lines with evenly distributed spaces
            while j < len(words) and total_length + 1 + len(words[j]) <= max_width:
                total_length += len(words[j]) + 1  
                j += 1
            
            space_count = max_width - total_length
            word_count = j - i  # number of words in the line, includes the last word
            if word_count > 1:
                base_space, extra_space = divmod(space_count, word_count - 1)  
            else:
                base_space = space_count
            
            line = ' '.join([words[k] + ' ' * (base_space if k < i + extra_space else 0) for k in range(i, j)]).rstrip()
            res.append(line)
            i = j
    return res

from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start] + ' ' * (max_width - total_length)
        
        num_spaces = end - start
        spaces_between_words = (max_width - total_length) // num_spaces
        extra_spaces = (max_width - total_length) % num_spaces
        
        line = words[start]
        for i in range(1, num_spaces + 1):
            line += ' ' * (spaces_between_words + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    current_length = len(words[0])
    
    for i in range(1, len(words)):
        word_length = len(words[i])
        if current_length + 1 + word_length > max_width:
            result.append(justify_line(words, start, i - 1, current_length))
            start = i
            current_length = word_length
        else:
            current_length += 1 + word_length
    
    # Handle the last line which should be left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result

def full_justify(words, max_width):
    res = []
    i = 0
    while i < len(words):
        total_length = len(words[i])
        j = i + 1
        # Handle case where a line has only one word by left-justifying it
        if j == len(words) or total_length + len(' '.join(words[i:j])) > max_width:  
            res.append(words[i] + ' ' * (max_width - total_length))
            i = j
        else:  # normal lines with evenly distributed spaces
            while j < len(words) and total_length + 1 + len(words[j]) <= max_width:
                total_length += len(words[j]) + 1  
                j += 1
            
            space_count = max_width - total_length
            word_count = j - i  # number of words in the line, includes the last word
            if word_count > 1:
                base_space, extra_space = divmod(space_count, word_count - 1)  
            else:
                base_space = space_count
            
            line = ' '.join([words[k] + ' ' * (base_space if k < i + extra_space else 0) for k in range(i, j)]).rstrip()
            res.append(line)
            i = j
    return res
