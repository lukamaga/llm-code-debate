from storage import StorageBackend

class TransactionException(Exception): pass

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transactions = []  # type: List[dict]

    def get(self, key: str) -> Union[str, None]:
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        if self.transactions:
            self.transactions[-1][key] = value
        else:
            self.storage.set(key, value)

    def delete(self, key: str) -> bool:
        if self.transactions and key in self.transactions[-1]:
            del self.transactions[-1][key]
            return True
        elif not self.transactions and self.storage.has(key):
            self.storage.delete(key)
            return True
        else:
            return False

    def begin(self) -> None:
        self.transactions.append(self.storage.snapshot())

    def commit(self) -> None:
        if not self.transactions:
            raise TransactionException('No active transaction to commit')

        for key, value in self.transactions[-1].items():
            self.storage.set(key, value)

        self.transactions.pop()

    def rollback(self) -> None:
        if not self.transactions:
            raise TransactionException('No active transaction to rollback')

        snapshot = self.transactions.pop()
        self.storage.restore(snapshot)

class Database:
    def __init__(self):
        self._storage = StorageBackend()
        self._transactions = []

    def get(self, key: str) -> str | None:
        for transaction in reversed(self._transactions):
            if key in transaction:
                return transaction[key]
        return self._storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self._transactions:
            self._storage.set(key, value)
        else:
            self._transactions[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self._transactions:
            return self._storage.delete(key)
        for transaction in reversed(self._transactions):
            if key in transaction:
                del transaction[key]
                return True
        return False

    def begin(self) -> None:
        self._transactions.append({})

    def commit(self) -> None:
        if not self._transactions:
            raise Exception("No active transaction")
        transaction = self._transactions.pop()
        for key, value in transaction.items():
            self._storage.set(key, value)

    def rollback(self) -> None:
        if not self._transactions:
            raise Exception("No active transaction")
        self._transactions.pop()

    def has(self, key: str) -> bool:
        for transaction in reversed(self._transactions):
            if key in transaction:
                return True
        return self._storage.has(key)

    def keys(self) -> list[str]:
        keys = set()
        for transaction in self._transactions:
            keys.update(transaction.keys())
        keys.update(self._storage.keys())
        return list(keys)

        if self._transactions:
            self._transactions[-1].clear()
        else:
            self._storage.clear()

    def snapshot(self) -> dict[str, str]:
        snapshot = self._storage.snapshot()
        for transaction in reversed(self._transactions):
            snapshot.update(transaction)
        return snapshot

        self._transactions.clear()

from storage import StorageBackend

class Database:
    def __init__(self):
        self._storage = StorageBackend()
        self._transactions = []

    def get(self, key: str) -> str | None:
        for transaction in reversed(self._transactions):
            if key in transaction:
                return transaction[key]
        return self._storage.get(key)

    def set(self, key: str, value: str) -> None:
        if not self._transactions:
            self._storage.set(key, value)
        else:
            self._transactions[-1][key] = value

    def delete(self, key: str) -> bool:
        if not self._transactions:
            return self._storage.delete(key)
        for transaction in reversed(self._transactions):
            if key in transaction:
                del transaction[key]
                return True
        return False

    def begin(self) -> None:
        self._transactions.append({})

    def commit(self) -> None:
        if not self._transactions:
            raise Exception("No active transaction")
        transaction = self._transactions.pop()
        for key, value in transaction.items():
            self._storage.set(key, value)

    def rollback(self) -> None:
        if not self._transactions:
            raise Exception("No active transaction")
        self._transactions.pop()

        for transaction in reversed(self._transactions):
            if key in transaction:
                return True
        return self._storage.has(key)

    def keys(self) -> list[str]:
        keys = set()
        for transaction in self._transactions:
            keys.update(transaction.keys())
        keys.update(self._storage.keys())
        return list(keys)

        else:
