        else:
            self.transaction_stack[-1][key] = value

        if not self.transaction_stack:
            return self.storage.delete(key)
        elif key in self.transaction_stack[-1]:
            del self.transaction_stack[-1][key]
            return True
        else:

        self.transaction_stack.append(self.storage.snapshot())

    def commit(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')
        else:
            self.storage.restore(self.transaction_stack.pop())

    def rollback(self) -> None:
        if not self.transaction_stack:
            raise Exception('No active transaction')
        else:

from storage import StorageBackend

class Database:
    def __init__(self):
        self.storage = StorageBackend()
        self.transactions = []

    def get(self, key: str) -> str | None:
        return self.storage.get(key)

    def set(self, key: str, value: str) -> None:
        self.transactions.append(('set', key, value))
        self.storage.set(key, value)

    def delete(self, key: str) -> bool:
        if self.storage.has(key):
            self.transactions.append(('delete', key))
            self.storage.delete(key)
            return True
        return False

    def begin(self) -> None:
        self.transactions.append(('begin',))

    def commit(self) -> None:
        if not self.transactions or self.transactions[-1][0] != 'begin':
            raise Exception("No active transaction")
        for operation in self.transactions[:-1]:
            if operation[0] == 'set':
                self.storage.set(operation[1], operation[2])
            elif operation[0] == 'delete':
                self.storage.delete(operation[1])
        del self.transactions[-1]

    def rollback(self) -> None:
        if not self.transactions or self.transactions[-1][0] != 'begin':
            raise Exception("No active transaction")
        while self.transactions and self.transactions[-1][0] != 'begin':
            operation = self.transactions.pop()
            if operation[0] == 'set':
                self.storage.delete(operation[1])
            elif operation[0] == 'delete':
                self.storage.set(operation[1], '')
        del self.transactions[-1]
