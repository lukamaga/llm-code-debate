            self.graph[v] = {}
        self.graph[u][v] = weight

    def shortest_path(self, start: str, end: str) -> tuple[float, list[str]]:

                if node not in visited and distances[node] < min_distance:
                    min_distance = distances[node]
                    current_node = node

            visited.add(current_node)

            for neighbor, weight in self.graph[current_node].items():
                distance = distances[current_node] + weight
                if distance < distances[neighbor]:
                    distances[neighbor] = distance

        current_node = end

            for node, weight in self.graph[current_node].items():
                if distances[node] == distances[current_node] - weight:
                    path.insert(0, node)
                    current_node = node
                    break

        return (distances[end], path)

                distance = distances[current_node] + weight
                if distance < distances[neighbor]:
                    distances[neighbor] = distance

        return {node: distances[node] for node in self.graph}

                    stack.append(neighbor)

import heapq
from typing import List, Tuple, Dict, Union

class WeightedGraph:
    def __init__(self):
        # adjacency list representation
        self.graph = {}

    def add_edge(self, u: str, v: str, weight: float) -> None:
        if u not in self.graph:
            self.graph[u] = {v: weight}
        else:
            self.graph[u][v] = weight

    def shortest_path(self, start: str, end: str) -> Tuple[Union[int, float], List[str]]:
        heap = [(0, start)]  # cost to reach start node is 0
        distances = {node: float('infinity') for node in self.graph}
        distances[start] = 0
        paths = {start: []}

        while heap:
            (curr_distance, curr_vertex) = heapq.heappop(heap)

            if curr_distance > distances[curr_vertex]:
                continue  # we already have a shorter path to this vertex

            for neighbor, weight in self.graph.get(curr_vertex, {}).items():
                distance = curr_distance + weight
                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    paths[neighbor] = paths[curr_vertex] + [neighbor]
                    heapq.heappush(heap, (distance, neighbor))

        return (distances[end], paths.get(end, [])) if end in distances else (float('inf'), [])

    def all_shortest_from(self, start: str) -> Dict[str, float]:
        return self.shortest_path(start, None)[0]

    def has_path(self, start: str, end: str) -> bool:
        visited = set()
        stack = [start]

        while stack:
            vertex = stack.pop()
            if vertex == end:
                return True
            visited.add(vertex)
            for neighbor in self.graph.get(vertex, {}):
                if neighbor not in visited and neighbor not in stack:
                    stack.append(neighbor)
        return False

        distances = {node: float('inf') for node in self.graph}

from heapq import heappop, heappush
from collections import defaultdict

class WeightedGraph:
    def __init__(self):
        self.graph = defaultdict(dict)

    def add_edge(self, u: str, v: str, weight: float) -> None:
        if v not in self.graph[u]:
            self.graph[u][v] = weight

    def shortest_path(self, start: str, end: str) -> tuple[float, list[str]]:
        distances = {node: float('inf') for node in self.graph}
        distances[start] = 0
        heap = [(0, start)]
        prev_nodes = {node: None for node in self.graph}

        while heap:
            current_distance, current_node = heappop(heap)

            if current_node == end:
                path = []
                while current_node is not None:
                    path.insert(0, current_node)
                    current_node = prev_nodes[current_node]
                return (distances[end], path)

            for neighbor, weight in self.graph[current_node].items():
                distance = current_distance + weight
                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    prev_nodes[neighbor] = current_node
                    heappush(heap, (distance, neighbor))

        return (float('inf'), [])

    def all_shortest_from(self, start: str) -> dict[str, float]:
        distances = {node: float('inf') for node in self.graph}
        distances[start] = 0
        heap = [(0, start)]
        prev_nodes = {node: None for node in self.graph}

        while heap:
            current_distance, current_node = heappop(heap)

            for neighbor, weight in self.graph[current_node].items():
                distance = current_distance + weight
                if distance < distances[neighbor]:
                    distances[neighbor] = distance
                    prev_nodes[neighbor] = current_node
                    heappush(heap, (distance, neighbor))

        return distances

    def has_path(self, start: str, end: str) -> bool:
        visited = set()
        stack = [start]

        while stack:
            node = stack.pop()
            if node == end:
                return True
            if node not in visited:
                visited.add(node)
                for neighbor in self.graph[node]:
                    stack.append(neighbor)

        return False
