class SchemaValidator:
    def validate(self, data: any, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True, errors=[])

        if schema.type == SchemaType.STRING:
            if not isinstance(data, str):
                result.errors.append(ValidationError(path='', message='Data is not a string'))
        elif schema.type == SchemaType.NUMBER or schema.type == SchemaType.INTEGER:
            if not isinstance(data, (int, float)):
                result.errors.append(ValidationError(path='', message='Data is not a number'))

            if schema.minimum is not None and data < schema.minimum:
                result.errors.append(ValidationError(path='', message=f'Data is less than {schema.minimum}'))

            if schema.maximum is not None and data > schema.maximum:
                result.errors.append(ValidationError(path='', message=f'Data is greater than {schema.maximum}'))
        elif schema.type == SchemaType.BOOLEAN:
            if not isinstance(data, bool):
                result.errors.append(ValidationError(path='', message='Data is not a boolean'))
        elif schema.type == SchemaType.OBJECT:
            if not isinstance(data, dict):
                result.errors.append(ValidationError(path='', message='Data is not an object'))

            for field in schema.required:
                if field not in data:
                    result.errors.append(ValidationError(path=field, message='Field is required'))

            for field, value in data.items():
                if field in schema.properties:
                    sub_result = self.validate(value, schema.properties[field])
                    sub_result.path = f'{field}.{sub_result.path}'
                    result.errors.extend(sub_result.errors)
        elif schema.type == SchemaType.ARRAY:
            if not isinstance(data, list):
                result.errors.append(ValidationError(path='', message='Data is not an array'))

            for i, item in enumerate(data):
                sub_result = self.validate(item, schema.items)
                sub_result.path = f'{i}.{sub_result.path}'
                result.errors.extend(sub_result.errors)
        elif schema.type == SchemaType.NULL:
            if data is not None:
                result.errors.append(ValidationError(path='', message='Data is not null'))
        else:
            raise ValueError('Invalid schema type')

        return result

            return ValidationResult(False, [ValidationError('', 'Expected an integer')])
        elif schema.type == SchemaType.BOOLEAN and not isinstance(data, bool):
            return ValidationResult(False, [ValidationError('', 'Expected a boolean')])
        elif schema.type == SchemaType.NULL and data is not None:
            return ValidationResult(False, [ValidationError('', 'Expected null')])
        elif schema.type == SchemaType.ARRAY:
            if not isinstance(data, list):
                errors.append(ValidationError('', 'Expected an array'))
            else:
                for i, item in enumerate(data):
                    result = self.validate(item, schema.items)
                    if not result.valid:
                        errors += [ValidationError(f'{i}.{e.path}', e.message) for e in result.errors]

            if not isinstance(data, dict):

                        errors += [ValidationError(f'{prop_name}.{e.path}', e.message) for e in result.errors]

                        errors.append(ValidationError('', f'Missing required field {required_field}'))
        return ValidationResult(len(errors) == 0, errors)

        elif type == SchemaType.NUMBER and not (isinstance(data, int) or isinstance(data, float)):
            return False
        elif type == SchemaType.INTEGER and not isinstance(data, int):
            return False

            return self._validate_array(data, schema)
        else:
            return ValidationResult(valid=True)

    def _type_check(self, data: any, type: SchemaType) -> bool:
        if type == SchemaType.STRING and not isinstance(data, str):
            return False
        elif type == SchemaType.NUMBER and not (isinstance(data, int) or isinstance(data, float)):
            return False
        elif type == SchemaType.INTEGER and not isinstance(data, int):
            return False
        elif type == SchemaType.BOOLEAN and not isinstance(data, bool):
            return False
        elif type == SchemaType.NULL and data is not None:
            return False
        else:
            return True

    def _validate_object(self, data: dict, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True)
        
        for field in schema.required:
            if field not in data:
                result.errors.append(ValidationError(path='', message=f'Missing required field {field}'))
                result.valid = False
        
        for key, value_schema in schema.properties.items():
            if key in data:
                sub_result = self.validate(data[key], value_schema)
                if not sub_result.valid:
                    result.errors.extend(sub_result.errors)
                    result.valid = False
            else:
                result.errors.append(ValidationError(path='', message=f'Missing required field {key}'))
                result.valid = False
        
        return result

    def _validate_array(self, data: list, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True)
        
        for i, item in enumerate(data):
            sub_result = self.validate(item, schema.items)
            if not sub_result.valid:
                result.errors.extend(sub_result.errors)
                result.valid = False

from schema_types import Schema, SchemaType, ValidationResult, ValidationError

class SchemaValidator:
    def validate(self, data: any, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True, errors=[])

        if schema.type == SchemaType.STRING:
            if not isinstance(data, str):
                result.errors.append(ValidationError(path='', message='Data is not a string'))
        elif schema.type == SchemaType.NUMBER or schema.type == SchemaType.INTEGER:
            if not isinstance(data, (int, float)):
                result.errors.append(ValidationError(path='', message='Data is not a number'))

            if schema.minimum is not None and data < schema.minimum:
                result.errors.append(ValidationError(path='', message=f'Data is less than {schema.minimum}'))

            if schema.maximum is not None and data > schema.maximum:
                result.errors.append(ValidationError(path='', message=f'Data is greater than {schema.maximum}'))
        elif schema.type == SchemaType.BOOLEAN:
            if not isinstance(data, bool):
                result.errors.append(ValidationError(path='', message='Data is not a boolean'))
        elif schema.type == SchemaType.OBJECT:
            if not isinstance(data, dict):
                result.errors.append(ValidationError(path='', message='Data is not an object'))

            for field in schema.required:
                if field not in data:
                    result.errors.append(ValidationError(path=field, message='Field is required'))

            for field, value in data.items():
                if field in schema.properties:
                    sub_result = self.validate(value, schema.properties[field])
                    sub_result.path = f'{field}.{sub_result.path}'
                    result.errors.extend(sub_result.errors)
        elif schema.type == SchemaType.ARRAY:
            if not isinstance(data, list):
                result.errors.append(ValidationError(path='', message='Data is not an array'))

            for i, item in enumerate(data):
                sub_result = self.validate(item, schema.items)
                sub_result.path = f'{i}.{sub_result.path}'
                result.errors.extend(sub_result.errors)
        elif schema.type == SchemaType.NULL:
            if data is not None:
                result.errors.append(ValidationError(path='', message='Data is not null'))
        else:
            raise ValueError('Invalid schema type')

        return result

from schema_types import Schema, SchemaType, ValidationResult, ValidationError

class SchemaValidator:
    def validate(self, data: any, schema: Schema) -> ValidationResult:
        if not self._type_check(data, schema.type):
            return self._error(f'Invalid type for {schema.type.value}')
        
        if schema.type == SchemaType.OBJECT:
            return self._validate_object(data, schema)
        elif schema.type == SchemaType.ARRAY:
            return self._validate_array(data, schema)
        else:
            return ValidationResult(valid=True)

    def _type_check(self, data: any, type: SchemaType) -> bool:
        if type == SchemaType.STRING and not isinstance(data, str):
            return False
        elif type == SchemaType.NUMBER and not (isinstance(data, int) or isinstance(data, float)):
            return False
        elif type == SchemaType.INTEGER and not isinstance(data, int):
            return False
        elif type == SchemaType.BOOLEAN and not isinstance(data, bool):
            return False
        elif type == SchemaType.NULL and data is not None:
            return False
        else:
            return True

    def _validate_object(self, data: dict, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True)
        
        for field in schema.required:
            if field not in data:
                result.errors.append(ValidationError(path='', message=f'Missing required field {field}'))
                result.valid = False
        
        for key, value_schema in schema.properties.items():
            if key in data:
                sub_result = self.validate(data[key], value_schema)
                if not sub_result.valid:
                    result.errors.extend(sub_result.errors)
                    result.valid = False
            else:
                result.errors.append(ValidationError(path='', message=f'Missing required field {key}'))
                result.valid = False
        
        return result

    def _validate_array(self, data: list, schema: Schema) -> ValidationResult:
        result = ValidationResult(valid=True)
        
        for i, item in enumerate(data):
            sub_result = self.validate(item, schema.items)
            if not sub_result.valid:
                result.errors.extend(sub_result.errors)
                result.valid = False
        
        return result

    def _error(self, message: str) -> ValidationResult:
        return ValidationResult(valid=False, errors=[ValidationError(path='', message=message)])
