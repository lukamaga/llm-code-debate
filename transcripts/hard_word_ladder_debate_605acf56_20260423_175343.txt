from collections import deque
from typing import List

def ladder_length(begin_word: str, end_word: str, word_list: List[str]) -> int:
    if end_word not in word_list or not begin_word or not end_word or not word_list:
        return 0
    
    word_set = set(word_list)
    queue = deque([(begin_word, 1)])
    visited = set([begin_word])
    
    while queue:
        current_word, level = queue.popleft()
        
        if current_word == end_word:
            return level
        
        for i in range(len(current_word)):
            for char in 'abcdefghijklmnopqrstuvwxyz':
                next_word = current_word[:i] + char + current_word[i+1:]
                
                if next_word in word_set and next_word not in visited:
                    queue.append((next_word, level + 1))
                    visited.add(next_word)
    
    return 0

from collections import deque
def ladder_length(begin_word, end_word, word_list):
    wordList = set(word_list) # Converting list to a set for O(1) lookup time
    queue = deque([[begin_word]]) 
    while queue:
        path = queue.popleft()
        w = path[-1]
        if w == end_word:
            return len(path) # Return the length of the path, i.e., number of words in shortest transformation sequence
        for i in range(len(w)):

                new_word = w[:i] + c + w[i+1:] # Generate all possible words by changing each character of the current word
                if new_word in wordList: 
                    wordList.remove(new_word) # Remove from set to avoid revisiting it and save time
                    queue.append(path + [new_word]) # Append this new word to path and continue BFS
    return 0 # If no such sequence exists, return 0
