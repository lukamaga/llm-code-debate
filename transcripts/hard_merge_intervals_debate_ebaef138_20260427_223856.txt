class IntervalManager:
    def __init__(self):
        self.intervals = []

    def add_interval(self, start: int, end: int) -> None:
        if not self.intervals:
            self.intervals.append((start, end))
        else:
            last_interval = self.intervals[-1]
            if start <= last_interval[1]:  # Overlap check
                self.intervals[-1] = (last_interval[0], max(last_interval[1], end))
            else:
                self.intervals.append((start, end))

    def get_merged(self) -> list[tuple[int, int]]:
        merged = []
        for interval in self.intervals:
            if not merged or interval[0] > merged[-1][1]:  # No overlap, append
                merged.append(interval)
            else:  # Overlap, merge
                merged[-1] = (merged[-1][0], max(merged[-1][1], interval[1]))
        return sorted(merged)

    def is_covered(self, point: int) -> bool:
        for interval in self.intervals:
            if interval[0] <= point <= interval[1]:
                return True
        return False

    def min_hitting_set(self) -> list[int]:
        points = set()
        for interval in self.get_merged():
            for p in range(interval[0], interval[1] + 1):
                points.add(p)
        return sorted(points)

class IntervalManager:
    def __init__(self):
        self.intervals = []

    def add_interval(self, start: int, end: int) -> None:
        # Add the new interval and sort all intervals
        self.intervals.append((start, end))
        self.intervals.sort()

        merged = [list(self.intervals[0])]
        for current in self.intervals:
            if current[0] <= merged[-1][1]:  # If the intervals overlap
                merged[-1][1] = max(merged[-1][1], current[1])  # Merge them by extending the last interval
            else:
                merged.append(list(current))  # Otherwise add a new one
        self.intervals = merged

    def get_merged(self) -> list[tuple[int, int]]:
        return self.intervals

    def is_covered(self, point: int) -> bool:
        for start, end in self.intervals:
            if start <= point <= end:  # If the point lies within an interval
                return True
        return False

    def min_hitting_set(self) -> list[int]:
        points = []
        for start, end in sorted(self.intervals):
            if points and start <= points[-1] <= end:  # If the last added point lies within this interval
                continue  # We don't need to add another one
            points.append(end)  # Otherwise we can use the end of this interval
        return points
