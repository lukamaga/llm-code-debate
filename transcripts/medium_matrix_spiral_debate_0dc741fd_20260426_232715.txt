def spiral_order(matrix):
    m, n = len(matrix), len(matrix[0])
    result = []
    dr, dc = 0, 1

    for _ in range(m * n):
        r, c = dr, dc
        while 0 <= r < m and 0 <= c < n:
            result.append(matrix[r][c])
            matrix[r][c] = None
            r += dr
            c += dc

        if r == m:
            r -= 1
            dr *= -1
        elif c == n:
            c -= 1
            dc *= -1

    return result
