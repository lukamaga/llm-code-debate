from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start] + ' ' * (max_width - total_length)
        
        num_spaces = end - start
        spaces_between_words = (max_width - total_length) // num_spaces
        extra_spaces = (max_width - total_length) % num_spaces
        
        line = words[start]
        for i in range(1, num_spaces + 1):
            line += ' ' * (spaces_between_words + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    total_length = len(words[0])
    
    for i in range(1, len(words)):
        word_length = len(words[i])
        if total_length + 1 + word_length > max_width:
            result.append(justify_line(words, start, i - 1, total_length))
            start = i
            total_length = word_length
        else:
            total_length += 1 + word_length
    
    # Last line is left-justified
    last_line = ' '.join(words[start:])
    last_line += ' ' * (max_width - len(last_line))
    result.append(last_line)
    
    return result

# Example usage:
words = ["This", "is", "an", "example", "of", "text", "justification."]

            formatted_lines.append(word)

        # If the length of the current line plus the word does not exceed the max width, add the word to it
        else:
            formatted_lines[-1] += " " + word
    
    # Return the list of formatted lines
    return formatted_lines

from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end):
        if start == end:
            return words[start] + ' ' * (max_width - len(words[start]))
        
        total_length = sum(len(word) for word in words[start:end])
        num_spaces = end - start
        spaces_between_words, extra_spaces = divmod(max_width - total_length, num_spaces)
        
        line = words[start]
        for i in range(1, num_spaces + 1):
            line += ' ' * (spaces_between_words + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    total_length = len(words[0])
    
    for i in range(1, len(words)):
        word_length = len(words[i])
        if total_length + 1 + word_length > max_width:
            result.append(justify_line(words, start, i - 1))
            start = i
            total_length = word_length
        else:
            total_length += 1 + word_length
    
    # Last line is left-justified
    last_line = ' '.join(words[start:])
    last_line += ' ' * (max_width - len(last_line))
    result.append(last_line)
    
    return result

from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start] + ' ' * (max_width - total_length)
        
        num_spaces = end - start
        spaces_between_words = (max_width - total_length) // num_spaces
        extra_spaces = (max_width - total_length) % num_spaces
        
        line = words[start]
        for i in range(1, num_spaces + 1):
            line += ' ' * (spaces_between_words + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    total_length = len(words[0])
    
    for i in range(1, len(words)):
        word_length = len(words[i])
        if total_length + 1 + word_length > max_width:
            result.append(justify_line(words, start, i - 1, total_length))
            start = i
            total_length = word_length
        else:
            total_length += 1 + word_length
    
    # Last line is left-justified
    last_line = ' '.join(words[start:])
    last_line += ' ' * (max_width - len(last_line))
    result.append(last_line)
    
    return result
