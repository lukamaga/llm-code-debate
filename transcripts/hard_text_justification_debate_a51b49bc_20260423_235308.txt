from typing import List

def full_justify(words: List[str], max_width: int) -> List[str]:
    def justify_line(words, start, end, total_length):
        if start == end:
            return words[start] + ' ' * (max_width - total_length)
        
        num_words = end - start
        spaces_to_distribute = max_width - total_length
        base_spaces = spaces_to_distribute // (num_words - 1)
        extra_spaces = spaces_to_distribute % (num_words - 1)
        
        line = words[start]
        for i in range(1, num_words):
            line += ' ' * (base_spaces + (1 if i <= extra_spaces else 0)) + words[start + i]
        
        return line
    
    result = []
    start = 0
    total_length = 0
    
    for i, word in enumerate(words):
        if total_length + len(word) + (i - start) > max_width:
            result.append(justify_line(words, start, i, total_length))
            start = i
            total_length = len(word)
        else:
            total_length += len(word)
    
    # Handle the last line which should be left-justified
    last_line = ' '.join(words[start:])
    result.append(last_line + ' ' * (max_width - len(last_line)))
    
    return result

def full_justify(words, max_width):

    i = 0
    while i < len(words):
        total_letters = len(words[i])
        j = i + 1
        while j < len(words) and (total_letters + len(words[j]) + j - i - 1 < max_width):
            total_letters += len(words[j])
            j += 1
            
        num_spaces = max_width - total_letters
        if j == len(words) or j - i == 1: # last line or only one word in current line
            line = " ".join(words[i:j]).ljust(max_width)
            result.append(line[:max_width])
        else: 
            num_spaces_between = num_spaces // (j - i - 1)
            num_extra_spaces = num_spaces % (j - i - 1)
            line = ""
            for k in range(i, j):
                if k == j - 1: # last word
                    line += words[k]
                else:

                    if num_extra_spaces > 0:
                        spaces += " "
                        num_extra_spaces -= 1

            result.append(line)
