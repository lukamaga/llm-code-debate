        i, j = 0, 0

        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                arr[i + j] = left[i]
                i += 1
            else:
                arr[i + j] = right[j]
                j += 1
                inv_count += len(left) - i

        arr[i + j:] = left[i:]
        arr[i + j:] = right[j:]
